( define ( domain jantar )
	( :requirements :strips :typing :equality )
	( :types )
	( :predicates
		( maos-limpas )
		( jantar )
		( silencio )
		( presente )
		( lixo )
	)

	( :action cozinhar
		:parameters ( )
		:precondition ( maos-limpas )
		:effect ( jantar )
	)

	( :action embrulhar
		:parameters ( )
		:precondition ( silencio )
		:effect ( presente )
	)

	( :action carregar-lixo
		:parameters ( )
		:precondition ( lixo )
		:effect ( and
			( not ( lixo ) )
			( not ( maos-limpas ) )
		)
	)

	( :action triturar-lixo
		:parameters ( )
		:precondition ( lixo )
		:effect ( and
			( not ( lixo ) )
			( not ( silencio ) )
		)
	)
)
